GitHub থেকে একটি Python ফাইল ডাউনলোড এবং চালানো
import requests

# GitHub থেকে raw URL
url = 'https://raw.githubusercontent.com/username/repository-name/branch-name/inst_loader.py'

response = requests.get(url)
if response.status_code == 200:
    with open('inst_loader.py', 'wb') as file:
        file.write(response.content)
    print("File downloaded successfully!")
    exec(open('inst_loader.py').read())
else:
    print("Failed to download the file")
